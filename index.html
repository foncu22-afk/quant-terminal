# ── Black-Scholes-Merton Engine ──────────────────────────────────
import numpy as np
from scipy.stats import norm
from scipy.optimize import brentq
from dataclasses import dataclass
from typing import Literal
import warnings; warnings.filterwarnings("ignore")

@dataclass
class GreeksResult:
    delta:  float
    gamma:  float
    theta:  float   # per calendar day
    vega:   float   # per 1% change in IV
    rho:    float
    price:  float
    iv:     float   # implied vol used

class BSMEngine:
    """
    Black-Scholes-Merton pricer with exact greeks.
    Supports European options. For American options,
    use binomial_price() with early-exercise detection.
    """

    def _d1_d2(self, S, K, T, r, q, sigma):
        """Compute d1/d2 with dividend yield q."""
        d1 = (np.log(S / K) + (r - q + 0.5 * sigma**2) * T) / (sigma * np.sqrt(T))
        d2 = d1 - sigma * np.sqrt(T)
        return d1, d2

    def price(self, S: float, K: float, T: float,
              r: float, q: float, sigma: float,
              flag: Literal["C", "P"]) -> float:
        """
        S: spot price | K: strike | T: time to expiry (years)
        r: risk-free rate | q: dividend yield | sigma: IV
        """
        if T <= 0: return max(0, S - K) if flag == "C" else max(0, K - S)
        d1, d2 = self._d1_d2(S, K, T, r, q, sigma)
        disc = np.exp(-r * T)
        if flag == "C":
            return S * np.exp(-q * T) * norm.cdf(d1) - K * disc * norm.cdf(d2)
        else:
            return K * disc * norm.cdf(-d2) - S * np.exp(-q * T) * norm.cdf(-d1)

    def greeks(self, S, K, T, r, q, sigma, flag) -> GreeksResult:
        d1, d2 = self._d1_d2(S, K, T, r, q, sigma)
        pdf_d1 = norm.pdf(d1)
        sqrt_T  = np.sqrt(T)
        disc    = np.exp(-r * T)
        eq_disc = np.exp(-q * T)

        # Delta
        delta = (eq_disc * norm.cdf(d1) if flag == "C"
                 else -eq_disc * norm.cdf(-d1))

        # Gamma (same for calls and puts)
        gamma = eq_disc * pdf_d1 / (S * sigma * sqrt_T)

        # Theta (per calendar day)
        common = (-(S * eq_disc * pdf_d1 * sigma) / (2 * sqrt_T))
        if flag == "C":
            theta = (common - r * K * disc * norm.cdf(d2)
                     + q * S * eq_disc * norm.cdf(d1)) / 365
        else:
            theta = (common + r * K * disc * norm.cdf(-d2)
                     - q * S * eq_disc * norm.cdf(-d1)) / 365

        # Vega (per 1% move in vol)
        vega = S * eq_disc * pdf_d1 * sqrt_T / 100

        # Rho (per 1% move in rates)
        if flag == "C":
            rho = K * T * disc * norm.cdf(d2) / 100
        else:
            rho = -K * T * disc * norm.cdf(-d2) / 100

        return GreeksResult(
            delta=round(delta, 6), gamma=round(gamma, 6),
            theta=round(theta, 6), vega=round(vega, 6),
            rho=round(rho, 6),   price=self.price(S,K,T,r,q,sigma,flag),
            iv=sigma
        )

    def implied_vol(self, market_price, S, K, T, r, q, flag,
                    tol=1e-6, max_iter=200) -> float:
        """Brent-Dekker IV solver — robust for deep ITM/OTM."""
        intrinsic = max(0, S - K) if flag == "C" else max(0, K - S)
        if market_price <= intrinsic + 1e-8: return np.nan
        try:
            return brentq(
                lambda v: self.price(S,K,T,r,q,v,flag) - market_price,
                1e-6, 10.0, xtol=tol, maxiter=max_iter
            )
        except ValueError:
            return np.nan

    def binomial_price(self, S, K, T, r, q, sigma, flag,
                       steps=500) -> float:
        """
        Cox-Ross-Rubinstein binomial tree.
        Handles American early-exercise (puts especially).
        Optimal for equity options with dividends.
        """
        dt   = T / steps
        disc = np.exp(-r * dt)
        u    = np.exp(sigma * np.sqrt(dt))
        d    = 1 / u
        pu   = (np.exp((r - q) * dt) - d) / (u - d)  # risk-neutral prob
        pd   = 1 - pu

        # Terminal stock prices (vectorised)
        j    = np.arange(steps + 1)
        ST   = S * (u ** (steps - j)) * (d ** j)

        # Terminal payoff
        V    = np.maximum(0, ST - K) if flag == "C" else np.maximum(0, K - ST)

        # Backward induction with early-exercise check
        for i in range(steps - 1, -1, -1):
            Snode = S * (u ** (i - np.arange(i + 1))) * (d ** np.arange(i + 1))
            hold  = disc * (pu * V[:-1] + pd * V[1:])
            if flag == "C": exercise = np.maximum(0, Snode - K)
            else:           exercise = np.maximum(0, K - Snode)
            V = np.maximum(hold, exercise)  # American: take max

        return float(V[0])


# ── Batch processor for OPRA feed ──────────────────────────────
def process_opra_batch(ticks: list[dict], spot: float,
                        r=0.053, q=0.014) -> list[dict]:
    """
    Vectorised greeks computation for a full option chain tick.
    ticks: list of {strike, expiry_days, flag, last_price, bid, ask}
    Returns enriched dicts with greeks + IV.
    """
    engine = BSMEngine()
    results = []
    for tick in ticks:
        T    = tick["expiry_days"] / 365
        mid  = (tick["bid"] + tick["ask"]) / 2
        iv   = engine.implied_vol(mid, spot, tick["strike"], T, r, q, tick["flag"])
        if np.isnan(iv): continue
        g = engine.greeks(spot, tick["strike"], T, r, q, iv, tick["flag"])
        results.append({**tick, "iv": iv, "delta": g.delta,
                        "gamma": g.gamma, "theta": g.theta,
                        "vega": g.vega, "price_bsm": g.price})
    return results

"""
Volatility Skew & Surface Calculator
─────────────────────────────────────
Input : JSON array from OPRA feed or broker API
Output: SVI-calibrated vol surface + skew metrics

JSON schema expected:
  {"underlying": "SPX", "spot": 5800,
   "options": [
     {"strike": 5600, "expiry_days": 30, "flag": "P",
      "bid": 12.5, "ask": 13.1, "oi": 8420},
     ...
   ]}
"""
import json, numpy as np, pandas as pd
from scipy.optimize import minimize, differential_evolution
from scipy.interpolate import RectBivariateSpline
from dataclasses import dataclass, field
from typing import Optional
from .bsm import BSMEngine

@dataclass
class SVIParams:
    # Raw SVI: w(k) = a + b*(ρ*(k-m) + sqrt((k-m)^2 + σ^2))
    a: float   # minimum total variance
    b: float   # ATM vol slope
    rho: float # correlation (skew param, -1 < ρ < 1)
    m: float   # ATM log-moneyness
    sigma: float # vol of vol (smile curvature)
    expiry_days: int = 0
    rmse: float = 0.0  # calibration error

class VolSkewCalculator:
    """
    Compute IV skew from option chain JSON.
    Fits SVI model per expiry, then interpolates across
    the full surface using bivariate spline.
    """

    def __init__(self, r=0.053, q=0.014):
        self.r = r
        self.q = q
        self.bsm = BSMEngine()

    def from_json(self, payload: str | dict) -> dict:
        """Main entry-point. Returns full surface dict."""
        data = json.loads(payload) if isinstance(payload, str) else payload
        spot = data["spot"]
        df   = pd.DataFrame(data["options"])
        return self.compute_surface(df, spot)

    def compute_surface(self, df: pd.DataFrame, spot: float) -> dict:
        """
        1. Strip IVs from market quotes
        2. Fit SVI per expiry
        3. Compute skew metrics (25Δ skew, RR, BF)
        4. Build vol surface grid
        """
        # 1. Compute IVs
        df["T"]  = df["expiry_days"] / 365
        df["mid"] = (df["bid"] + df["ask"]) / 2
        df["k"]  = np.log(df["strike"] / (spot * np.exp(
                           (self.r - self.q) * df["T"])))  # log-moneyness
        df["iv"] = df.apply(lambda row: self.bsm.implied_vol(
            row["mid"], spot, row["strike"], row["T"],
            self.r, self.q, row["flag"]), axis=1)
        df = df.dropna(subset=["iv"])
        df = df[(df["iv"] > 0.01) & (df["iv"] < 5.0)]  # sanity filter

        # 2. Fit SVI per expiry
        expiries = sorted(df["expiry_days"].unique())
        svi_fits: list[SVIParams] = []
        for exp in expiries:
            sub = df[df["expiry_days"] == exp].copy()
            if len(sub) < 6: continue
            params = self._fit_svi(sub["k"].values, sub["iv"].values, int(exp))
            if params: svi_fits.append(params)

        # 3. Compute skew metrics per expiry
        skew_metrics = self._compute_skew_metrics(df, spot, expiries)

        # 4. Build surface grid (strikes x expiries)
        surface_grid = self._build_surface(svi_fits, spot)

        return {
            "spot":          spot,
            "svi_params":    [p.__dict__ for p in svi_fits],
            "skew_metrics":  skew_metrics,
            "surface_grid":  surface_grid,
            "atm_term_struct": self._atm_term_structure(svi_fits, spot),
        }

    def _svi_total_var(self, k, a, b, rho, m, sigma):
        """Raw SVI parameterisation → total variance w(k,T)."""
        z = k - m
        return a + b * (rho * z + np.sqrt(z**2 + sigma**2))

    def _fit_svi(self, k_arr, iv_arr, expiry_days) -> Optional[SVIParams]:
        """
        Calibrate SVI to market IVs via differential evolution
        (global opt) + Nelder-Mead polish.
        No-arbitrage butterfly constraint enforced.
        """
        T = expiry_days / 365
        w_market = iv_arr**2 * T   # market total variance

        def objective(p):
            a, b, rho, m, sigma = p
            w = self._svi_total_var(k_arr, a, b, rho, m, sigma)
            return np.mean((np.sqrt(np.maximum(w, 0) / T) - iv_arr) ** 2)

        bounds = [(0, 0.5), (0, 1), (-0.99, 0.99),
                  (-0.5, 0.5), (1e-4, 1.0)]
        try:
            # Global search
            res_de = differential_evolution(objective, bounds, seed=42,
                                            maxiter=500, tol=1e-8)
            # Polish with Nelder-Mead
            res = minimize(objective, res_de.x, method="Nelder-Mead",
                           options={"xatol":1e-9,"fatol":1e-9,"maxiter":2000})
            a, b, rho, m, sigma = res.x
            rmse = np.sqrt(objective(res.x)) * 100   # in vol points
            return SVIParams(a, b, rho, m, sigma, expiry_days, round(rmse, 4))
        except Exception:
            return None

    def _compute_skew_metrics(self, df, spot, expiries) -> list[dict]:
        """
        25-delta skew metrics per expiry:
        - Risk Reversal  (RR) = IV_25C - IV_25P  ← directional bias
        - Butterfly      (BF) = (IV_25C + IV_25P)/2 - IV_ATM  ← smile
        - Skew slope     = dIV/d(moneyness) at ATM
        """
        metrics = []
        for exp in expiries:
            sub = df[df["expiry_days"] == exp]
            T   = exp / 365
            # ATM vol
            atm_calls = sub[(sub["flag"]=="C") &
                            (abs(sub["k"]) < 0.05)]
            if atm_calls.empty: continue
            iv_atm = atm_calls.iloc[(atm_calls["k"].abs()).argsort()[:1]]["iv"].mean()
            # 25-delta approximation: k ≈ ±0.5σ√T (Δ≈0.25)
            k_25 = 0.5 * iv_atm * np.sqrt(T)
            iv_25c = sub[(sub["flag"]=="C") &
                         (abs(sub["k"] - k_25) < 0.1)]["iv"].mean()
            iv_25p = sub[(sub["flag"]=="P") &
                         (abs(sub["k"] + k_25) < 0.1)]["iv"].mean()
            if np.isnan(iv_25c) or np.isnan(iv_25p): continue
            # Expected move (1-sigma, ATM straddle approximation)
            expected_move_pct = iv_atm * np.sqrt(T) * 100
            metrics.append({
                "expiry_days":     exp,
                "iv_atm":          round(iv_atm * 100, 2),
                "risk_reversal_25": round((iv_25c - iv_25p) * 100, 2),
                "butterfly_25":     round(((iv_25c+iv_25p)/2 - iv_atm) * 100, 2),
                "expected_move_pct": round(expected_move_pct, 2),
                "expected_move_abs": round(spot * expected_move_pct / 100, 1),
            })
        return metrics

    def _build_surface(self, svi_fits: list[SVIParams], spot) -> dict:
        """Generate vol surface on a grid of strikes x expiries."""
        if not svi_fits: return {}
        k_grid = np.linspace(-0.4, 0.4, 40)            # log-moneyness
        strikes_grid = spot * np.exp(k_grid)
        surface = {}
        for fit in svi_fits:
            T = fit.expiry_days / 365
            w = self._svi_total_var(k_grid, fit.a, fit.b, fit.rho, fit.m, fit.sigma)
            iv_grid = np.sqrt(np.maximum(w, 0) / T) * 100   # in %
            surface[fit.expiry_days] = {
                "strikes": strikes_grid.round(1).tolist(),
                "ivs":     iv_grid.round(2).tolist(),
                "rmse":    fit.rmse,
            }
        return surface

    def _atm_term_structure(self, svi_fits, spot) -> list[dict]:
        """ATM vol term structure for VIX-style roll analysis."""
        return [
            {"expiry_days": p.expiry_days,
             "iv_atm_pct": round(np.sqrt(p.a + p.b * p.sigma * np.sqrt(1 - p.rho**2)
                                          / np.sqrt(p.expiry_days/365)) * 100, 2)}
            for p in svi_fits if p.expiry_days > 0
        ]


# ── Max Pain Calculator ─────────────────────────────────────────
def max_pain(chain_df: pd.DataFrame) -> tuple[float, pd.DataFrame]:
    """
    Calculate max pain level: strike minimising total
    option value decay (calls + puts OI weighted).
    Returns (max_pain_strike, strike_level_df).
    """
    strikes = sorted(chain_df["strike"].unique())
    pain = []
    for K_test in strikes:
        calls = chain_df[chain_df["flag"]=="C"]
        puts  = chain_df[chain_df["flag"]=="P"]
        call_pain = (calls["oi"] * np.maximum(0, K_test - calls["strike"])).sum()
        put_pain  = (puts["oi"]  * np.maximum(0, puts["strike"]  - K_test)).sum()
        pain.append({"strike": K_test, "total_pain": call_pain + put_pain})
    df = pd.DataFrame(pain)
    mp  = df.loc[df["total_pain"].idxmin(), "strike"]
    return float(mp), df

        self.active.setdefault(channel, []).append(ws)
    def disconnect(self, ws: WebSocket, channel: str):
        if channel in self.active: self.active[channel].discard(ws)
    async def broadcast(self, channel: str, data: dict):
        dead = []
        for ws in self.active.get(channel, []):
            try: await ws.send_json(data)
            except: dead.append(ws)
        for ws in dead: self.active[channel].discard(ws)

# ── Real-time Greeks Stream ─────────────────────────────────────
@app.websocket("/ws/greeks/{underlying}")
async def greeks_stream(ws: WebSocket, underlying: str):
    """
    Streams live greeks for all options on `underlying`.
    Subscribes to Redis channel opra:{underlying}
    and recomputes greeks on each tick.
    Latency target: <20ms tick-to-client.
    """
    await mgr.connect(ws, f"greeks:{underlying}")
    redis = aioredis.Redis(connection_pool=redis_pool)
    pubsub = redis.pubsub()
    await pubsub.subscribe(f"opra:{underlying}")
    try:
        async for message in pubsub.listen():
            if message["type"] != "message": continue
            tick = json.loads(message["data"])
            spot = tick["spot"]
            r    = tick.get("risk_free", 0.053)
            q    = tick.get("div_yield", 0.014)
            T    = tick["expiry_days"] / 365
            iv   = engine.implied_vol(
                       (tick["bid"]+tick["ask"])/2,
                       spot, tick["strike"], T, r, q, tick["flag"])
            if iv and not (iv != iv):   # nan check
                g = engine.greeks(spot, tick["strike"], T, r, q, iv, tick["flag"])
                await ws.send_json({
                    "type":   "greeks",
                    "strike": tick["strike"],
                    "expiry": tick["expiry_days"],
                    "flag":   tick["flag"],
                    "iv_pct": round(iv * 100, 2),
                    "delta":  g.delta, "gamma": g.gamma,
                    "theta":  g.theta, "vega":  g.vega,
                    "ts":     tick["ts"],
                })
    except WebSocketDisconnect:
        mgr.disconnect(ws, f"greeks:{underlying}")
    finally:
        await pubsub.unsubscribe()
